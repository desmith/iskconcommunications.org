server {
	listen 80; # managed by Certbot
	listen 443 ssl; # managed by Certbot
	server_name www.iskconcommunications.org
		.iskconcommunications.org
		.iskconcommunications.co.uk
		.iskconcommunications.us
		.iskconcommunications.info
		.iskconcommunications.net
		.iskconcommunications.com
	;

  root /var/www/iskconcommunications.org;

  include conf.d/common.conf;

	# Redirect non-https traffic to https
	if ($scheme != "https") {
		return 301 https://$host$request_uri;
	} # managed by Certbot

  if ($http_host !~ "^www.iskconcommunications.org$"){
    rewrite ^/(.*)$ "http://www.iskconcommunications.org/$1" redirect;
  }

  location / {
    try_files $uri $uri/ /index.html;
  }

	ssl_certificate /etc/letsencrypt/live/www.iskconcommunications.org/fullchain.pem; # managed by Certbot
	ssl_certificate_key /etc/letsencrypt/live/www.iskconcommunications.org/privkey.pem; # managed by Certbot
	include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
	ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
